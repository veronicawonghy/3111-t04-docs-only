package comp3111.popnames;

/**
 * Class {@code NameRecord} is the class for Task One 
 * and Four Application.
 *
 * @author	Ceci Au
 * @version	1.0
 */
public class NameRecord {
	private final String name;
    private final String gender;
    private final int number;
    private final int year;
    private final String percentage;
 
    /**
     * Constructs a new NameRecommendationService.
     */
    public NameRecord(String name, String gender, int number, int year) {
        this.name = name;
        this.gender = gender;
        this.number = number;
        this.year = year;
        this.percentage = String.format("%.2f%%",(double)number*100/AnalyzeNames.getTotalBirthByGender(year, gender));
    }
 
    /**
     * Returns the data member {@code name}.
     * <p>
     * @return  a string of name.
     */
    public String getName() {
        return name;
    }
        
    /**
     * Returns the data member {@code gender}.
     * <p>
     * @return  a string of gender.
     */
    public String getGender() {
        return gender;
    }
    
    /**
     * Returns the data member {@code number}.
     * <p>
     * @return  an int of number/occurrence.
     */
    public int getNumber() {
        return number;
    }
    
    /**
     * Returns the data member {@code year}.
     * <p>
     * @return  an int of year.
     */
    public int getYear() {
        return year;
    }
    
    /**
     * Returns the data member {@code percentage}.
     * <p>
     * @return  an string of percentage, in two decimal places.
     */
    public String getPercentage() {
        return percentage;
    }
}
