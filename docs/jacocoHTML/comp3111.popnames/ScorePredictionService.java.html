	public static String isValidName(String name) {
		if (name.trim().isEmpty()) return "Please input a name! \n";
		if (name.length() < 2 || name.length() > 15) return "Please input a name that has 2-15 characters!\n";
		String regex = "[A-Za-z\\s]+";

	/**
	 * Get the summary of result of simple algorithm
	 * <p>
	 * The calculation is referenced from <a href="https://en.wikipedia.org/wiki/Numerology#Pythagorean_system">https://en.wikipedia.org/wiki/Numerology#Pythagorean_system</a>
	 * </p>
	 * @param name Any name
	 * @return Life Path number
	 */
	public static int getLifePathNumber(String name) {
		String temp = name.toLowerCase();
		int sum;
		while (true) {
			sum = 0;
			for(int i = 0; i < temp.length(); i++)
			{
				char c = temp.charAt(i);
				switch(c) { // Source: https://en.wikipedia.org/wiki/Numerology
					case 'a': case 'j': case 's':
						sum += 1;
						break;
					case 'b': case 'k': case 't':
						sum += 2;
						break;
					case 'c': case 'l': case 'u':
						sum += 3;
						break;
					case 'd': case 'm': case 'v':
						sum += 4;
						break;
					case 'e': case 'n': case 'w':
						sum += 5;
						break;
					case 'f': case 'o': case 'x':
						sum += 6;
						break;
					case 'g': case 'p': case 'y':
						sum += 7;
						break;
					case 'h': case 'q': case 'z':
						sum += 8;
						break;
					case 'i': case 'r':
						sum += 9;
						break;
					case ' ':
						sum += 0;
						break;
					default: 
						sum += Integer.parseInt(String.valueOf(c));
						break;
				}
			}
			if (sum <= 9) {
				return sum;
			}
			temp = String.valueOf(sum);
		}
	}
	
	/**
	 * Get the summary of detailed compatibility 
	 * <p>
	 * The detailed algorithm is composed of two parts - numerology and astrology.
	 * If horoscope of the person to be matched is not sure, the score will be calculated purely by life path number compatibility.
	 * Otherwise, the score will be calculated by 60% of horoscope compatibility and 40% of life path number compatibility.
	 * </p>
	 * @return Summary of detailed compatibility 
	 */
	public String getDetailedCompatibility() {
		String result = "";
		int userHoroscopeVal = this.userHoroscope.ordinal();
		int mateHoroscopeVal = this.mateHoroscope.ordinal();
		
		// Get numerology score
		int userLifeNum = this.getLifePathNumber(this.userName);
		int mateLifeNum = this.getLifePathNumber(this.mateName);
		String numeroLogyScore = NumerologyCompatibility[userLifeNum-1][mateLifeNum-1];
		float numeroLogyScoreValue = Float.parseFloat(numeroLogyScore.replace("%","")) / 100;
		
		// If the mate horoscope is not sure, only use the numerology score
		result  += "The Compatibility score is "+numeroLogyScore+"!\n";
		String numeroLogyExplanation = "Your life number is "+userLifeNum+" and "+this.mateName+" life number is "+mateLifeNum+". ";
		if (numeroLogyScore == "100%") {
			numeroLogyExplanation += "In numerology, your life number is highly compatible with the number of your partner. "
					+ "You guys tend to be riding a similar wavelength. When you meet your partner, things instantly click."
					+ "These connections flow with ease and feel destined and filled with magic.\n";
		} else if (numeroLogyScore == "50%") {
			numeroLogyExplanation += "In numerology, your life number is neutral with the number of your partner. "
					+ "That means, you two can either be very compatible or not, or that could perhaps be entirely neutral, holding no real charge.\n ";
		} else {
			numeroLogyExplanation += "In numerology, your life number is not very compatible with the number of your partner. "
					+ "You guys may have different core values and potential to clash. It's not written in stone that you will experience conflict with these numbers, "
					+ "but you will be challenged to see the world from a new perspective. Though this may feel seemingly negative, these pairings actually have much to teach and learn from each other.\n";
		}
		if(this.mateHoroscope.name() == "NOT_SURE") {
			result += numeroLogyExplanation;
			return result;
		};
		// If the mate horoscope is known, 40% numerology and 60% horoscope score
		float horoscopeScoreValue = 0;
		String horsocpeScore = HoroscopeCompatibility[userHoroscopeVal][mateHoroscopeVal];
		horoscopeScoreValue = Float.parseFloat(horsocpeScore.replace("%","")) / 100;
		
		double scoreValue = horoscopeScoreValue*0.6+ numeroLogyScoreValue*0.4;
		result = "The Compatibility score is "+String.format("%.0f%%",scoreValue*100)+"!\n";
		
		result += numeroLogyExplanation;
		
		if(scoreValue<0.5) {
			result += "\nWhile in astrology, your horoscopes are not very compatible.. But don't worry too much and check our advice below :)";
		}
		else if(scoreValue>0.9) result +="\nWhile in astrology, your horoscopes are highly compatible! You guys are born to be together!";
		else result +="\nWhile in astrology, your horoscopes are quite compatible! Check our advice below :) ";
		result += numeroLogyExplanation;
		return result;
	}
